TAT Segments

The turnaround time is broken into five segments: | Segment | Definition | Measures | |———|————|———-| | 1: Pre-lab | Requisition → Received | Collection & transport time | | 2: In-lab | Received → Released | Laboratory processing time | | 3: Post-lab | Released → Acknowledged | Clinical uptake time | | 4: End-to-end | Requisition → Released | Total lab responsibility | | 5: Total | Requisition → Acknowledged | Complete clinical cycle |

Process Flow: Median TAT by Category

Code 
# Calculate medians for completed samples
flow_data <- tat_analysis |>
 filter(!is.na(kvitteret)) |>
 group_by(kategori_label) |>
 summarise(
   seg1 = median(seg1_prelab, na.rm = TRUE),
   seg2 = median(seg2_inlab, na.rm = TRUE),
   seg3 = median(seg3_postlab, na.rm = TRUE),
   .groups = "drop"
 ) |>
 pivot_longer(cols = starts_with("seg"), names_to = "segment", values_to = "minutes") |>
 mutate(
   segment_label = case_match(
     segment,
     "seg1" ~ "Pre-lab",
     "seg2" ~ "In-lab",
     "seg3" ~ "Post-lab"
   ),
   segment_label = factor(segment_label, levels = c("Pre-lab", "In-lab", "Post-lab"))
 )

# Format time helper
format_time <- function(mins) {
 if(is.na(mins)) return(NA_character_)
 if (mins < 60) {
   paste0(round(mins), "m")
 } else if (mins < 24 * 60) {
   paste0(round(mins / 60, 1), "h")
 } else {
   paste0(round(mins / (24 * 60), 1), "d")
 }
}

ggplot(flow_data, aes(x = segment_label, y = minutes, fill = kategori_label)) +
 geom_col(position = "dodge", width = 0.7) +
 geom_text(
   aes(label = sapply(minutes, format_time)),
   position = position_dodge(width = 0.7),
   vjust = -0.5,
   size = 3
 ) +
 scale_fill_brewer(palette = "Set2", name = "Category") +
 scale_y_continuous(
   labels = function(x) sapply(x, format_time),
   expand = expansion(mult = c(0, 0.15))
 ) +
 labs(
   x = "TAT Segment",
   y = "Median Time",
   title = "Process Flow: Median TAT by Segment",
   subtitle = "Requisition → Received → Released → Acknowledged"
 ) +
 theme(legend.position = "bottom")

Pending Samples

Code 
pending <- tat_analysis |>
 filter(is.na(besvaret) | is.na(kvitteret)) |>
 mutate(
   status = case_when(
     is.na(besvaret) ~ "Pending result",
     is.na(kvitteret) ~ "Pending acknowledgment"
   ),
   age_hours = as.numeric(difftime(Sys.time(), afsendt, units = "hours"))
 ) |>
 select(proevenr, kategori_label, status, afsendt, age_hours)

if (nrow(pending) > 0) {
 pending |>
   arrange(desc(age_hours)) |>
   gt() |>
   tab_header(title = "Pending Samples") |>
   fmt_datetime(columns = afsendt, date_style = "iso", time_style = "iso") |>
   fmt_number(columns = age_hours, decimals = 1, pattern = "{x}h") |>
   cols_label(
     proevenr = "Sample ID",
     kategori_label = "Category",
     status = "Status",
     afsendt = "Requisition Time",
     age_hours = "Age"
   )
} else {
 cat("No pending samples.")
}
